# Правила

> Правила совпадают с расширенной формой Бэкуса-Наура

```js
А - нетерминал
```

```js
`Б` - текст
```

```js
А, Б - конкатенация (АБ)
```

```js
А | Б - выбор (А или Б)
```

```js
[ А ] - необязательный элемент (ничего или А)
```

```js
{ Б } - повторение (ничего, Б, ББ...)
```

```js
( А Б ) - группировка
```

```js
А = Б
  | В

А = Б | В
```

```js
А = Б
  , В

А = Б, В
```

# Лексемы

## Сокращения

```js
цифра - 0-9
символ - любой символ
буква - А-я и A-z
завершение_конструкции - `---` или `===`
перенос_строки - `\n` или `;`
```


## Основные лексемы

```js
число = целое_число | вещественное_число
```

```js
целое_число = цифра, { цифра }
```

```js
вещественное_число = целое_число, `,`, целое_число
```

```js
строка = `"`, { символ }, `"`
```

```js
список = `%(`, [ операция, { `;`, операция } ], `)%`
```

```js
идентификатор = ( `_` | буква ), { буква | цифра }
```

## Ключевые слова

> Для выделения, ключевые слова начинаются с `!`

```js
!вернуть    = `вернуть`    | `return`
!прервать   = `прервать`   | `break`
!пропустить = `пропустить` | `skip`
!включить   = `включить`   | `include`
!удалить    = `удалить`    | `delete`
```

<!--
```js
!ничего = `ничего` | `pass`
```
-->

```js
!и   = `и`   | `and`
!или = `или` | `or`
!не  = `не`  | `not`
```

```js
!если  = `если`  | `if`
!то    = `то`    | `then`
!иначе = `иначе` | `else`
```

```js
!проверить = `проверить` | `check`
!при       = `при`       | `on`
```

```js
!для   = `для`   | `for`
!от    = `от`    | `from`
!до    = `до`    | `to`
!через = `через` | `after`
!пока  = `пока`  | `while`
```

```js
!функция = `функция` | `function`
!класс   = `класс`   | `class`
```


# Граматика

```js
программа = { операция, перенос_строки }
```

```js
операция = простая_операция
         | сложная_операция
         | комментарий
```

```js
комментарий = `!!`, { символ }
            | `!*`, { символ | перенос_строки }, `*!`
```

```js
простая_операция = вызов
                 | включение
                 | удаление
                 | возврат
                 | контекстный_переход
                 | присваивание
                 | выражение
```

```js
сложная_операция = конструкция_если
                 | конструкция_проверить
                 | конструкция_для
                 | конструкция_пока
                 | объявление_функции
                 | объявление_класса
```


# Простые оперции

```js
вызов = идентификатор, `(`, [ выражение, { `,`, выражение } ], `)`
```

```js
включение = !включить, строка
```

```js
удаление = !удалить, идентификатор
```

```js
возврат = !вернуть, выражение
```

```js
контекстный_переход = !прервать
                    | !пропустить
```

<!-- | !ничего -->

```js
присваивание = идентификатор, `=`, ( выражение | присваивание )
```

```js
выражение = унарная_операция
          | бинарная_операция
          | операнд
```

```js
бинарная_операция = степень_корень
                  | умножение_деление
                  | сумма_разность
                  | логическая_операция
```

```js
логическая_операция = арифметическое_сравнение
                    | арифметическое_сравнение, !и, арифметическое_сравнение
                    | арифметическое_сравнение, !или, арифметическое_сравнение
```

```js
арифметическое_сравнение = сумма_разность
                         | сумма_разность, оператор_сравнения, сумма_разность
                         , [оператор_сравнения, сумма_разность]
```

```js
оператор_сравнения = `==` | `!=` | `\/` | `/\` | `\/=` | `/\=`
```

```js
сумма_разность = умножение_деление, [`+` | `-`, умножение_деление]
```

```js
умножение_деление = степень_корень, [`*` | `/`, степень_корень]
```

```js
степень_корень = унарная_операция, [`**` | `///`, унарная_операция]
```

```js
унарная_операция = (`++` | `--`), идентификатор
                 | идентификатор, (`++` | `--`)
                 | `-`, операнд
                 | !не, выражение
```

```js
операнд = идентификатор
        | число
        | строка
        | `(`, выражение, `)`
```

# Сложные оперции

```js
блок = перенос_строки, { операция, перенос_строки }
```

```js
конструкция_если = !если, выражение, !то, блок
                 , { !иначе, !если, выражение, !то, блок }
                 , [ !иначе, блок ], завершение_конструкции
                 | !если, выражение, !то, операция
                 , { !иначе, !если, выражение, !то, операция }
                 , [ !иначе, операция ]
```

```js
конструкция_проверить = !проверить, идентификатор, [ оператор_сравнения ]
                      , !при, [ оператор_сравнения ], идентификатор
                      , { ( !и | !или ), [ оператор_сравнения ], идентификатор }
                      , блок
                      , { !при, [ оператор_сравнения ], идентификатор
                      , { ( !и | !или ), [ оператор_сравнения ], идентификатор }
                      , блок }
                      , [ !иначе, блок ], завершение_конструкции
```

```js
конструкция_для = !для, идентификатор, !от, сумма_разность
                , !до, сумма_разность, [ !через, сумма_разность ]
                , блок, [ !иначе, блок ], завершение_конструкции
                | !для, идентификатор, !от, сумма_разность
                , !до, сумма_разность, [ !через, сумма_разность ]
                , `:`, операция
                | !для, идентификатор, !из, ( идентификатор | список )
                , блок, [ !иначе, блок ], завершение_конструкции
                | !для, идентификатор, !из, ( идентификатор | список )
                , `:`, операция
```

```js
конструкция_пока = !пока, выражение, блок
                , [ !иначе, блок ], завершение_конструкции
                | !пока, идентификатор, выражение, `:`, операция
```

```js
аргумент = идентификатор | присвоение
```

```js
объявление_функции = !функция, [ идентификатор ], `(`, [ аргумент, { `;`, аргумент } ] `)`
                   , ( ( блок, завершение_конструкции ) | ( `:`, операция ) )
```

```js
объявление_класса = !класс, [ идентификатор ], `(`, [ аргумент, { `;`, аргумент } ] `)`
                  , { объявление_метода }, завершение_конструкции
```

```js
объявление_метода = !функция, идентификатор, `(`, идентификатор, { `;`, аргумент } `)`
                  , ( ( блок, завершение_конструкции ) | ( `:`, операция ) )
